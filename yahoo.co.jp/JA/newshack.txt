<script type="text/javascript">
var app = angular.module('newshack', ['ngResource']);
app.factory('articleList', function($resource) {
return $resource (
'/newshack/:category/more.json',
{category: '@category'}
);
});
app.controller('listController', ['$scope', '$http', '$log', 'articleList', function ($scope, $http, $log, articleList) {
$scope.listRow = parseInt($('#list_row').val());
$scope.listColumn = 3;
$scope.list = [];
$scope.page = 1;
$scope.view = 0;
$scope.more = true;
$scope.ctg = $('#category').val();
$scope.wHeight = 0;
$scope.getMore = function() {
$scope.wHeight = $(window).scrollTop();
articleList.get({category: $scope.ctg}).$promise.then(function(result) {
var data = result.data;
var cnt = data.length;
var list = [];
var tmplist = [];
angular.forEach(data, function(value, key) {
if (key >= ($scope.listRow * $scope.listColumn)) {
tmplist.push(value);
if ((key % $scope.listColumn) == ($scope.listColumn-1)) {
// 各行の末尾
list.push(tmplist);
tmplist = [];
} else if ((key +1) == cnt) {
// 全ての記事の最後
list.push(tmplist);
tmplist = [];
}
}
});
$scope.list = list;
$scope.view = $scope.listRow * $scope.page ;
var showCnt = $scope.listRow * $scope.listColumn * ($scope.page + 1);
if (cnt <= showCnt) {
$scope.more = false;
}
$scope.page++;
});
}
$scope.$on('ngRepeatFinished', function(ngRepeatFinishedEvent) {
$log.info('dispHeight: ' + $scope.wHeight);
$(window).scrollTop($scope.wHeight);
});
}]);
app.directive('onFinishRender', function ($timeout) {
return {
restrict: 'A',
link: function (scope, element, attr) {
if (scope.$last === true) {
$timeout(function () {
scope.$emit('ngRepeatFinished');
});
}
}
}
});
</script>
